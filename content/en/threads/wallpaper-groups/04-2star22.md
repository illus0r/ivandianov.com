Reflections along both axes and 2-fold rotations on a centered rectangular lattice.

::: glsl editor classic collapsed
precision highp float;
uniform vec2 resolution;
uniform vec2 mouse;
uniform float time;
out vec4 o;

vec2 cmm(vec2 uv){
  if((int(floor(uv.y/2.))&1)==0){uv.x=1.-uv.x;}
  uv=abs(mod(uv,2.)-1.);
  return uv;
}

void main(){
  vec2 uv = (gl_FragCoord.xy * 2. - resolution) / 
    min(resolution.x, resolution.y) * 3. + .001;
    
  uv = cmm(uv);

  vec3 a=vec3(0,.85,.5),b=vec3(.9,0,.6),c=vec3(.95,.85,0);
  o.rgb+=mix(mix(a,b,uv.x),c,uv.y*1.4);
}
